REST API Principles
Fundamentals
🔗
Stateless
Each request must contain all information. Server stores no client state between requests.
🎯
Resource-Based
Resources are identified by URIs. Use nouns, not verbs: /users/{id}
⚙️
HTTP Methods
GET (read), POST (create), PUT/PATCH (update), DELETE (remove). Use correctly always.
📦
JSON Responses
Standard: Content-Type: application/json with consistent structure.
🛡️
Versioning
Use /api/v1/. Never break backward compatibility.
⚡
Caching
HTTP caching with ETag, Cache-Control, and Last-Modified headers.
🔒
Security
HTTPS only, rate limiting, input validation, proper CORS, Authorization headers.
📄
HATEOAS
Hypermedia links in responses let clients discover actions dynamically.

Python · Flask
from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)
db  = SQLAlchemy(app)

class User(db.Model):
    id    = db.Column(db.Integer, primary_key=True)
    name  = db.Column(db.String(100))
    email = db.Column(db.String(120), unique=True)

# GET /api/v1/users
@app.route('/api/v1/users', methods=['GET'])
def get_users():
    page  = request.args.get('page', 1, type=int)
    users = User.query.paginate(page=page, per_page=10)
    return jsonify({'data': [{'id':u.id,'name':u.name} for u in users.items], 'total':users.total})

# POST /api/v1/users
@app.route('/api/v1/users', methods=['POST'])
def create_user():
    data = request.get_json()
    user = User(name=data['name'], email=data['email'])
    db.session.add(user); db.session.commit()
    return jsonify({'id': user.id}), 201

# PUT /api/v1/users/<id>
@app.route('/api/v1/users/<int:uid>', methods=['PUT'])
def update_user(uid):
    user = User.query.get_or_404(uid)
    data = request.get_json()
    user.name = data.get('name', user.name)
    db.session.commit()
    return jsonify({'message': 'Updated'})

# DELETE /api/v1/users/<id>
@app.route('/api/v1/users/<int:uid>', methods=['DELETE'])
def delete_user(uid):
    user = User.query.get_or_404(uid)
    db.session.delete(user); db.session.commit()
    return jsonify({'message': 'Deleted'})

Node.js · Express
const express = require('express');
const { User } = require('./models');
const router = express.Router();

router.get('/', async (req, res) => {
  const { page=1, limit=10 } = req.query;
  const users = await User.findAndCountAll({ limit, offset:(page-1)*limit });
  res.json({ data: users.rows, total: users.count });
});

router.post('/', async (req, res) => {
  const user = await User.create(req.body);
  res.status(201).json(user);
});

router.put('/:id', async (req, res) => {
  const user = await User.findByPk(req.params.id);
  if (!user) return res.status(404).json({ error:'Not found' });
  await user.update(req.body); res.json(user);
});

router.delete('/:id', async (req, res) => {
  const user = await User.findByPk(req.params.id);
  if (!user) return res.status(404).json({ error:'Not found' });
  await user.destroy(); res.status(204).send();
});

module.exports = router;

package main

import ("encoding/json";"net/http";"strconv"
    "github.com/gorilla/mux";"gorm.io/gorm")

type User struct {
    gorm.Model
    Name  string `json:"name"`
    Email string `json:"email" gorm:"unique"`
}

func GetUsers(w http.ResponseWriter, r *http.Request) {
    var users []User
    page, _ := strconv.Atoi(r.URL.Query().Get("page"))
    if page == 0 { page = 1 }
    db.Offset((page-1)*10).Limit(10).Find(&users)
    json.NewEncoder(w).Encode(users)
}

func CreateUser(w http.ResponseWriter, r *http.Request) {
    var user User
    json.NewDecoder(r.Body).Decode(&user)
    db.Create(&user)
    w.WriteHeader(201)
    json.NewEncoder(w).Encode(user)
}

Authentication APIs
Security
🎫 JWT Authentication
Stateless tokens. Store on client, verify on server. Best for SPAs & mobile apps.
1
POST /auth/login → validate credentials
2
Server signs JWT with secret, returns token
3
Client sends Authorization: Bearer <token>
4
Server verifies signature, extracts claims
🔑 OAuth 2.0 / OpenID
Delegated authorization — "Login with Google/GitHub". Industry standard for third-party access.
1
Redirect user to provider endpoint
2
User authenticates → provider returns code
3
Exchange code for access_token
4
Use token to call protected APIs
🗝️ API Key Auth
Simple static tokens. Best for server-to-server or public APIs with rate limiting.
1
Generate cryptographically random key
2
Store hashed version in database
3
Client sends X-API-Key: <key>
4
Server hashes incoming key, compares
🍪 Session-Based Auth
Server-side sessions. Best for server-rendered web apps with Redis session store.
1
User submits login form
2
Server creates session in Redis/DB
3
Session ID sent as HTTP-only cookie
4
Cookie auto-sent on all subsequent requests

const jwt = require('jsonwebtoken');

// Middleware
const auth = (req, res, next) => {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'No token' });
  try { req.user = jwt.verify(token, process.env.SECRET); next(); }
  catch { res.status(401).json({ error: 'Invalid token' }); }
};

// Login
router.post('/login', async (req, res) => {
  const { email, password } = req.body;
  const user = await User.findOne({ where: { email }});
  if (!user || !await bcrypt.compare(password, user.password))
    return res.status(401).json({ error: 'Invalid credentials' });
  const token = jwt.sign({ id: user.id }, process.env.SECRET, { expiresIn: '15m' });
  res.json({ token });
});
PHP · Laravel Sanctum
class AuthController extends Controller {
    public function login(Request $request) {
        $request->validate(['email'=>'required|email','password'=>'required']);
        if (!Auth::attempt($request->only('email','password')))
            return response()->json(['msg'=>'Invalid credentials'], 401);
        $token = $request->user()->createToken('api')->plainTextToken;
        return response()->json(['token'=>$token]);
    }
    public function logout(Request $request) {
        $request->user()->tokens()->delete();
        return response()->json(['msg'=>'Logged out']);
    }
}

Go · JWT Middleware
var jwtSecret = []byte("super-secret-key")

func JWTMiddleware(next http.HandlerFunc) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        tokenStr := r.Header.Get("Authorization")[7:]
        token, err := jwt.Parse(tokenStr,
            func(t *jwt.Token) (interface{}, error) {
                return jwtSecret, nil
            })
        if err != nil || !token.Valid {
            http.Error(w, "Unauthorized", 401); return
        }
        next(w, r)
    }
}
